# --- Grad-CAM Fix for Swin Transformer ---
def generate_grad_cam(model, input_tensor, target_category):
    model.eval()
    # Fixed: Swin Transformer uses .features instead of .layers
    target_layer = model.backbone.features[-1]
    
    features_blobs = []
    gradients_blobs = []

    def hook_feature(module, input, output): features_blobs.append(output)
    def hook_gradient(module, grad_input, grad_output): gradients_blobs.append(grad_output[0])

    handle_f = target_layer.register_forward_hook(hook_feature)
    handle_g = target_layer.register_full_backward_hook(hook_gradient)

    output = model(input_tensor)
    loss = output[0, target_category]
    model.zero_grad()
    loss.backward()

    grads = gradients_blobs[0].cpu().data.numpy()
    feats = features_blobs[0].cpu().data.numpy()
    
    # Swin output is (B, H, W, C) or (B, L, C) - we treat as spatial
    weights = np.mean(grads, axis=(0, 1, 2))
    cam = np.zeros(feats.shape[1:3], dtype=np.float32)

    for i, w in enumerate(weights):
        cam += w * feats[0, :, :, i]

    handle_f.remove()
    handle_g.remove()
    return cam

import numpy as np
import matplotlib.pyplot as plt

import os

def get_gradcam(model, img_tensor, label):
    model.eval()
    # Targeting the last stage of features in Swin
    target_layer = model.backbone.features[-1]
    
    features, grads = [], []
    def save_f(m, i, o): features.append(o)
    def save_g(m, gi, go): grads.append(go[0])
    
    h1 = target_layer.register_forward_hook(save_f)
    h2 = target_layer.register_full_backward_hook(save_g)
    
    # Forward & Backward Pass
    out = model(img_tensor)
    model.zero_grad()
    out[0, label].backward()
    
    # Process features and gradients
    f = features[0].cpu().data.numpy()[0]
    g = grads[0].cpu().data.numpy()[0]
    weights = np.mean(g, axis=(0, 1))
    cam = np.dot(f, weights)
    
    # Post-processing for high-quality visualization
    cam = np.maximum(cam, 0) # ReLU
    cam = cv2.resize(cam, (224, 224))
    cam = (cam - np.min(cam)) / (np.max(cam) - np.min(cam) + 1e-10)
    
    h1.remove(); h2.remove()
    return cam

def generate_medical_visualization(model, idx, class_name):
    img, lbl = val_data[idx]
    img_t = img.unsqueeze(0).to(device)
    
    # 1. Generate Grad-CAM
    heatmap = get_gradcam(model, img_t, lbl) # Uses the function from Step 5.3
    
    # 2. Process Original for plotting
    orig = img.permute(1, 2, 0).numpy()
    orig = (orig - orig.min()) / (orig.max() - orig.min())
    orig_uint8 = (orig * 255).astype(np.uint8)
    
    # 3. Create Bounding Box from Heatmap (Threshold at 50% intensity)
    thresh = (heatmap * 255).astype(np.uint8)
    _, binary = cv2.threshold(thresh, 127, 255, cv2.THRESH_BINARY)
    contours, _ = cv2.findContours(binary, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    
    # Find the largest contour (the tumor)
    box_img = orig.copy()
    if contours:
        c = max(contours, key=cv2.contourArea)
        x, y, w, h = cv2.boundingRect(c)
        cv2.rectangle(box_img, (x, y), (x + w, y + h), (0, 1, 0), 2) # Green Box

    # 4. Plotting Row

    axes[0].set_title(f"Original MRI: {class_name}", fontsize=14)

    axes[1].set_title("Grad-CAM Heatmap", fontsize=14)
    axes[2].imshow(box_img)
    axes[2].set_title("Automated Tumor Localization", fontsize=14)
    
    for ax in axes: ax.axis('off')
    plt.tight_layout()

def generate_ieee_comparison(model, idx, class_name):

    heatmap = get_gradcam(model, img_t, lbl)
    
    # 2. Process Original MRI
    orig = img.permute(1, 2, 0).detach().cpu().numpy()
    orig = (orig - orig.min()) / (orig.max() - orig.min())
    
    # 3. Create Heatmap Overlay + Bounding Box
    heatmap_colored = plt.get_cmap('jet')(heatmap)[:, :, :3]
    overlay = (0.5 * orig + 0.5 * heatmap_colored)
    
    # Thresholding to find the peak tumor location
    thresh = (heatmap * 255).astype(np.uint8)
    _, binary = cv2.threshold(thresh, 150, 255, cv2.THRESH_BINARY)
    contours, _ = cv2.findContours(binary, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    
    if contours:
        c = max(contours, key=cv2.contourArea)
        x, y, w, h = cv2.boundingRect(c)
        # Bold White Box for maximum HD clarity
        cv2.rectangle(overlay, (x, y), (x + w, y + h), (1, 1, 1), 2)
